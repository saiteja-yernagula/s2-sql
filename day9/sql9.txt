constraints are rules that are applied to the columnsf
for data integrity
syntax
create table std (col datatype constraint);

null
Ensures that a column cannot have a NULL value.
create table std (id int, name varchar(20) not null);

insert into std (id,name) values (1,'john');//inserted
insert into std (id) values (2);//it will throw the error
insert into std (name) values ('jessy');//inserted

check 
Ensures that the values in a column meet a specified condition.
and it also accepts null.
create table std (id int, 
           name varchar(20) not null, age int check(age>18));

alter table std add column age int check(age>18);
insert into std (id,name,age) values (1,'jack',25);//inserted
insert into std (id,name,age) values (1,'jack',12);//error

unique
Ensures all values in a column are unique and allows null values.
create table std (id int, 
           name varchar(20), adharno int unique);

truncate table std;
alter table std add column adharno int unique;
insert into std (id,name,age,adharno) values (1,'jack',25,123);--inserted
insert into std (id,name,age,adharno) values (1,'jack',25,124);//inserted
insert into std (id,name,age,adharno) values (1,'jack',25,124);//error
insert into std (id,name,age,adharno) values (1,'jack',25,null);//inserted
