

logical operators
and
 select * from emp where sal<1500 and deptno=10;
 select * from emp where deptno>10 and job='manager';
or 
  select * from emp where deptno=10 or sal<1500;
  select * from emp where job='manager' or job='clerk';
  select * from emp where sal=800.50 or job='manager';
not
  select * from emp where not job='clerk';
  select * from emp where not deptno=10 and not deptno=20;

special operators
is null - it is used to whether the value is null or not
  select * from emp where comm is null;

is not null- it is used to whether the value is not null or null
   select * from emp where comm is not null;

in - it is used to whethere the records are within set of values
   or not
 select * from emp where deptno in (10,20,40);

not in
 select * from emp where deptno in (10,20,40);

between -it is used to display the records in a range
 select * from emp where sal between 1000 and 2000;

not between- it is used to display the records outside of a range
select * from emp where sal not between 1000 and 2000;

wildcard operators - it is used to match a specific pattern
while retrieving a data

like operator
%--for matching multiple characters
 select * from emp where ename like 's%';//starting with s
  select * from emp where ename like '%s';//ending with s
  select * from emp where ename like '%s%';//contain s

_ --for matching single character
  select * from emp where ename like '_ard';//ward
 select * from emp where ename like '___d'; //ford and ward
select * from emp where ename like '__N__';//jones
