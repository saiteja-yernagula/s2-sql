

JOINS
Joins in MySQL are used to retrieve data from multiple tables
 based on a related column
1. INNER JOIN
	Explanation:
The INNER JOIN keyword selects records that have matching values in both tables.
If there is no match, the rows are excluded from the result set.
This is the most commonly used join in SQL.
It is useful when we only need records that
   have corresponding entries in both tables.

syntax
select * from table1 t1 inner join
      table2 t2 on t1.refcol=t2.refcol;

select * from emp e inner join
               dept d on e.deptno=d.deptno;

get the details who are from location dallas
select * from emp e inner join dept d on
                  e.deptno=d.deptno where d.loc='dallas';

get the emp name and department they are working
select e.ename ,d.dname from emp e
                      join dept d on e.deptno=d.deptno;













1) Display student_id who registered under anitha
select e.student_id from trainers t join enrollments e on
     t.trainer_id=e.trainer_id where t.trainer_name='anita';

2) Display student_id who registered for python
select e.student_id from courses c join enrollments e on 
    c.course_id=e.course_id where c.course_name='python';

3) Display who was teaching the course having highest fees
select * from trainers t join courses c on 
      t.expertise=c.course_name order by c.fee desc limit 1;

4) Display sid,cid,cname,fees of each std
select e.student_id,e.course_id , c.course_name,c.fee
 from enrollments e join courses c on e.course_id=c.course_id;

5) Display studentid,total fees paid by each student
select e.student_id ,  sum(c.fee) from enrollments e join 
  courses c on e.course_id=c.course_id group by e.student_id;
