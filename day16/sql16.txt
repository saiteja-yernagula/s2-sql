1) Employees who work in the Sales department
select ename from emp where deptno=(select 
    deptno from dept where dname='sales');

2) Employees earning more than the average salary
select ename from emp where sal>(select
  avg(sal) from emp);

3) Employees who work in the same department as SMITH
select ename from emp where deptno=(select
    deptno from emp where ename='smith');

4) Highest paid employee
select ename from emp where sal=(select max(sal) 
from emp);

5) Employees who work in NEW YORK
select ename from emp where deptno=(select 
deptno from dept where loc='new york');

6) Employees whose salary is greater than FORD’s salary
select ename from emp where sal>(select sal 
from emp where ename='ford');
