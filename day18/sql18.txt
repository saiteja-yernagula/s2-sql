select * from courses where

tcl - transaction control language

create table std (id int,name varchar(20));

 insert into std values (1,'john');

 set autocommit=0;

 start transaction;
 insert into std values (2,'jessy');
 insert into std values (3,'jack');
 commit;

 start transaction;
 insert into std values (3,'ram');
 insert into std values (4,'ramesh');
 rollback;
 select * from std;
